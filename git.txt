git status

# создать коммит
git commit ""
git commit -m ""

# создать ветку
git branch название_ветки
git branch -d название_ветки

# перейти на ветку
git checkout название_ветки

# удалить ветку
git branch -d название_ветки

# изменить название ветки
git branch -m старое_имя_ветки новое_имя_ветки.

# Чтобы в ветку develop влить master
# Для этого нам надо сначала переключиться на ту ветку,
# в которую мы хотим вливать (develop), а затем уже «мержить».
git checkout develop
git merge master
